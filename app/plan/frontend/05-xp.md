## Functions

### Get XP Balance

```typescript
// hooks/useXpBalance.ts
import { useQuery } from '@tanstack/react-query';
import { useConnection } from '@solana/wallet-adapter-react';
import { PublicKey } from '@solana/web3.js';
import { getAssociatedTokenAddressSync } from '@solana/spl-token';

export function useXpBalance(wallet: PublicKey | null, xpMint: PublicKey | null) {
  const { connection } = useConnection();
  
  return useQuery({
    queryKey: ['xpBalance', wallet?.toBase58()],
    queryFn: async () => {
      if (!wallet || !xpMint) return 0;
      
      const ata = getAssociatedTokenAddressSync(
        xpMint,
        wallet,
        false,
        TOKEN_2022_PROGRAM_ID
      );
      
      try {
        const balance = await connection.getTokenAccountBalance(ata);
        return Number(balance.value.amount);
      } catch {
        return 0;
      }
    },
    enabled: !!wallet && !!xpMint,
    refetchInterval: 30000, // Refresh every 30 seconds
  });
}
```

### Get XP Mint from Config

```typescript
// hooks/useXpMint.ts
export function useXpMint() {
  const { program } = useProgram();
  const configPda = deriveConfigPda(PROGRAM_ID);
  
  return useQuery({
    queryKey: ['xpMint'],
    queryFn: async () => {
      const config = await program.account.config.fetch(configPda);
      return config.xpMint;
    },
  });
}
```

### XP Progress Bar

```typescript
// components/xp/XpProgressBar.tsx
interface XpProgressBarProps {
  currentXp: number;
  nextLevelXp: number;
  level: number;
}

export function XpProgressBar({ currentXp, nextLevelXp, level }: XpProgressBarProps) {
  const progress = (currentXp / nextLevelXp) * 100;
  
  return (
    <div className="xp-progress">
      <div className="level-badge">Level {level}</div>
      <div className="progress-bar">
        <div className="progress-fill" style={{ width: `${progress}%` }} />
      </div>
      <div className="progress-label">
        {currentXp} / {nextLevelXp} XP
      </div>
    </div>
  );
}
```

## Level Calculations

```typescript
// lib/levels.ts
export function calculateLevel(xp: number): number {
  // Each level requires progressively more XP
  // Level 1: 0-999, Level 2: 1000-2499, etc.
  if (xp < 1000) return 1;
  if (xp < 2500) return 2;
  if (xp < 5000) return 3;
  if (xp < 10000) return 4;
  if (xp < 20000) return 5;
  if (xp < 35000) return 6;
  if (xp < 55000) return 7;
  if (xp < 80000) return 8;
  if (xp < 120000) return 9;
  return 10;
}

export function getXpForLevel(level: number): number {
  const thresholds = [0, 1000, 2500, 5000, 10000, 20000, 35000, 55000, 80000, 120000];
  return thresholds[Math.min(level, thresholds.length - 1)];
}

export function getNextLevelXp(xp: number): number {
  const level = calculateLevel(xp);
  return getXpForLevel(level + 1);
}
```
